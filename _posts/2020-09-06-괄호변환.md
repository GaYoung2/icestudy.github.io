### 서우빈
```
def solution(p):
    if p=='':
        return p
    u, v = '', ''
    pos = 0
    for i in range(len(p)):
        pos += 1 if p[i]=='(' else -1
        u += p[i]
        if pos==0:
            v += p[i+1:]
            break
    if len(u)!= 0 and u[0]==')':
        return '(' + solution(v) + ')' + ''.join(['(' if i==')' else ')' for i in u[1:len(u)-1]])
    return u+solution(v)
```
