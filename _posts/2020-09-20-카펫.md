### 서우빈
```
def solution(brown, yellow):
    nums = []
    root = 0
    for i in range(1,yellow+1):
        if yellow%i==0:
            if i*i == yellow:
                root = i
                continue
            nums.append(i)
    if len(nums)==0:
        return [3,3]
    if (root+2)*2+2*root == brown:
        return [root+2,root+2]
    for h,w in zip(nums[0:len(nums)//2],sorted(nums[len(nums)//2:],reverse=True)):
        if h*2+(w+2)*2 == brown:
            return [w+2,h+2]
 ```
